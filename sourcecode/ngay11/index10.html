function greetWorld() {
            alert("Hello World!");
        }
         
        // Selecting button element
        var btn = document.getElementById("myBtn");
         
        // gắn sự kiện vào thẻ
        btn.addEventListener("click", greetWorld);
         
        // bỏ sự kiện khỏi thẻ
        btn.removeEventListener("click", greetWorld);
